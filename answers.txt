		self.age=age

		self.desig=desig

		print("Cop Details:")
		print("Name: ",self.name)
		print("Age: ",self.age)
		print("Work Experience: ",self.workexp)
		print("Designation: ",self.desig)


class Mission(Cop):
	basicObjective='prevent crime'
